def add_cement(n, costs):

    n.madd(
        "Bus",
        spatial.cement,
        location=spatial.cement.locations,
        carrier="cement",
        unit="t cement",
    )


    n.madd(
        "Load",
        spatial.cement,
        suffix=" cement demand",
        bus=spatial.cement.nodes,
        carrier="cement",
        p_set,
    )

    n.madd(
        "Link",
        bos0=spatial.nodes,
        bus1=spatial.ht_heat.nodes,
        bus2=spatial.cement.nodes,
        bus3="co2 atmosphere",
        carrier="cement production",
        efficiency=,
        efficiency1=,
        efficiency2=,
        lifetime=,
    )

    n.madd(
        "Link",
        bos0=spatial.nodes,
        bus1=spatial.ht_heat_cc.nodes,
        bus2=spatial.cement.nodes,
        bus3=spatial.co2.nodes,
        carrier="cement production cc",
        efficiency=,
        efficiency1=,
        efficiency2=,
        lifetime=,
        capital_cost=,
        p_nom_exendable=,
    )

    n.madd(
        "Link",
        bos0=spatial.nodes,
        bus1=spatial.ht_heat_oxy.nodes,
        bus2=spatial.cement.nodes,
        bus3=spatial.co2.nodes,
        carrier="cement production oxyfuel",
        efficiency=,
        efficiency1=,
        efficiency2=,
        lifetime=,
        capital_cost=,
        p_nom_exendable=,
    )

    # Adding High Temperetaure Heat buses

    n.madd(
        "Bus",
        spatial.ht_heat,
        location=spatial.ht_heat.locations,
        carrier="HT heat",
        unit="MWh",
    )
    n.madd(
        "Bus",
        spatial.ht_heat_cc,
        location=spatial.ht_heat_cc.locations,
        carrier="HT heat cc",
        unit="MWh",
    )
    n.madd(
        "Bus",
        spatial.ht_heat_oxy,
        location=spatial.ht_heat_oxy.locations,
        carrier="HT heat oxy",
        unit="MWh",
    )

    # Add Waste Bus

    n.madd(
        "Bus"
        spatial.waste,
        location=spatial.waste.locations,
        carrier="waste",
        unit="MWh"
    )
    n.madd(
        "Generator",
        bus=spatial.waste.nodes,
        p_nom_extendable=True,
        carrier="waste",
        marginal_cost=,
    )

    # High Temperatur Heat via oil

    n.madd(
        "Link",
        bus0=spatial.oil.nodes,
        bus1=spatial.ht_heat.nodes,
        bus2="co2 atmosphere",
        efficiency=1,
        efficiency2=,

    n.madd(
        "Link",
        bus0=spatial.oil.nodes,

        efficiency2=,

        "Link",
        bus0=spatial.oil.nodes,
        bus1=spatial.ht_heat_oxy.nodes,
        bus2=spatial.co2.nodes,
        efficiency=1,
        efficiency2=,

    n.madd(
        "Link",
        bus0=spatial.gas.nodes,
        bus1=spatial.ht_heat.nodes,
        bus2="co2 atmosphere",
        efficiency=1,
        efficiency2=,
    )

        "Link",
        bus0=spatial.gas.nodes,

        efficiency=1,

        bus0=spatial.coal.nodes,

        efficiency2=,

    n.madd(
        "Link",
        bus0=spatial.coal.nodes,
        bus1=spatial.ht_heat_oxy.nodes,
        bus2=spatial.co2.nodes,
        efficiency=1,
        efficiency2=,
    )

    # High Temperature Heat via Waste

    n.madd(
